def simulate(params: SimulationParams, rng: np.random.Generator) -> Distribution:
    """
    Agent-based stochastic polymer growth simulation.
    
    Model:
        - Growth: chain + 1 monomer with P = pg × monomer_ratio
        - Death: chain stops growing with P = pd × monomer_ratio
        - Vampiric: dead + living → coupled with P = f(lengths, exponents)
    
    Parameters from thesis.txt Table I.
    
    Args:
        params: Simulation parameters (validated)
        rng: NumPy random generator for reproducibility
        
    Returns:
        Distribution of living, dead, coupled chain lengths
        
    Example:
        >>> params = SimulationParams(time_sim=1000, ...)
        >>> rng = np.random.default_rng(42)
        >>> result = simulate(params, rng)
        >>> print(len(result.living))
    """
    params.validate()
    
    # Initialize
    living = np.ones(params.number_of_molecules)
    dead = np.array([])
    coupled = np.array([])
    
    # Run simulation
    for t in range(params.time_sim):
        living, dead, coupled = _simulation_step(living, dead, coupled, params, rng)
    
    return Distribution(living, dead, coupled)

def test_determinism():
    """Same seed → same output."""
    params = SimulationParams(100, 10000, 1000000, 0.72, 0.000084, 0.73, 0.41, 0.75, 0.24, True)
    
    rng1 = np.random.default_rng(42)
    dist1 = simulate(params, rng1)
    
    rng2 = np.random.default_rng(42)
    dist2 = simulate(params, rng2)
    
    assert np.array_equal(dist1.living, dist2.living)
    assert np.array_equal(dist1.dead, dist2.dead)

# src/polymer_growth/cli/main.py
import click
from pathlib import Path
from polymer_growth.api import run_simulation, fit_parameters

@click.group()
def cli():
    """Polymer Growth Simulation & Parameter Fitting."""
    pass

@cli.command()
@click.option('--config', type=click.Path(exists=True), required=True)
@click.option('--seed', type=int, required=True)
@click.option('--out', type=click.Path(), default='results/')
def simulate(config, seed, out):
    """Run single simulation with fixed parameters."""
    click.echo(f"Running simulation (seed={seed})...")
    result = run_simulation(config, seed)
    result.save(out)
    click.echo(f"✓ Results saved to {out}")

@cli.command()
@click.option('--target', type=click.Path(exists=True), required=True)
@click.option('--config', type=click.Path(exists=True), required=True)
@click.option('--seed', type=int, required=True)
@click.option('--out', type=click.Path(), default='results/')
def fit(target, config, seed, out):
    """Fit parameters to target distribution."""
    click.echo(f"Fitting to {target} (seed={seed})...")
    result = fit_parameters(target, config, seed, 
                           progress_callback=lambda g, c: click.echo(f"Gen {g}: {c}"))
    result.save(out)
    click.echo(f"✓ Best params: {result.best_params}")
